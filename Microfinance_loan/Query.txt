    BorrowerID VARCHAR2(12) PRIMARY KEY,

    VillageArea VARCHAR2(150),

CREATE TABLE LOAN_TBL (
    LoanID  VARCHAR2(15) PRIMARY KEY,
    BorrowerID VARCHAR2(12),
    ProductName VARCHAR2(100),
    PrincipalAmount   NUMBER(12,2),
    AnnualInterestRate  NUMBER(5,2),
    TermMonths  NUMBER(3),
    RepaymentFrequency VARCHAR2(20),
    DisbursementDate DATE,
    OutstandingPrincipal  NUMBER(12,2),
    Status VARCHAR2(20),
    CONSTRAINT fk_borrower
        FOREIGN KEY (BorrowerID)
        REFERENCES BORROWER_TBL(BorrowerID)
);

CREATE TABLE INSTALLMENT_TBL (
    Installment_ID   NUMBER(10) PRIMARY KEY,
    LoanID   VARCHAR2(15),
    InstallmentNo NUMBER(5),
    DueDate DATE,
    DueAmount NUMBER(12,2),
    PaidAmount  NUMBER(12,2),
    PaidDate DATE,
    Status VARCHAR2(20),
    CONSTRAINT fk_loan
        FOREIGN KEY (LoanID)
